
function drive(phone, message, web) {
  if (Array.isArray(phone)) {
    return phone.map((row, index) => drive(row[0], message[index][0]), web);
  }
  const link = phone;
  let result = link.slice(32,-18);
  return web === true
    ? `https://drive.google.com/uc?export=download&id=${result}`
    : `https://drive.google.com/uc?export=download&id=${result}`;
}
